-테이블 생성

create table user (
	user_seqno Integer primary Key auto_increment,
    user_id varchar(50) not null,
    user_pw varchar(50) not null,
    user_name varchar(50) not null,
    user_email varchar(50) not null,
	user_phone varchar(50) not null,
    user_address varchar(255) not null
);

create table board (
	bno Integer primary key	auto_increment,
    title varchar(50) not null,
    content text not null,
    writer varchar(50) not null,
    view_count int not null default 0,
    reg_date datetime not null default current_timestamp
);

create table comment (
	cno Integer primary key auto_increment,
    bno Integer not null,
    commenter varchar(50) not null,
    comment varchar(255) not null,
    reg_date datetime not null default current_timestamp
);

create table list(
list_seqno int primary key auto_increment,
list_no int not null,
list_img varchar(50) not null,
list_name varchar(50) not null,
list_s_price int not null,
list_o_price int not null,
list_desc text not null,
list_ex varchar(50) not null,
list_cateno int not null
);
